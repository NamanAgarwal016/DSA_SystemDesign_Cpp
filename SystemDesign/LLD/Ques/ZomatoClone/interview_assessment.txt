Q4: How would you extend the design to support split payment (e.g., 50% card, 50% netbanking) or add order retries/refunds?
Answer:

Split Payment:
Create a composite payment strategy:

class SplitPaymentStrategy : public PaymentStrategy {
private:
    vector<PaymentStrategy*> strategies;  // Multiple payment modes
    vector<int> amounts;  // Amounts for each mode
public:
    void addPaymentMode(PaymentStrategy* mode, int amount) {
        strategies.push_back(mode);
        amounts.push_back(amount);
    }
    void pay(int totalAmount) override {
        for(size_t i = 0; i < strategies.size(); i++) {
            strategies[i]->pay(amounts[i]);
        }
    }
};

Q5: How would you scale notification delivery (async queue, retries) and add logging/observability?
Answer:

Async Notification Queue:
Replace synchronous notification with an async queue:

class NotificationQueue {
private:
    queue<Order*> pendingNotifications;
    bool isProcessing;
public:
    void enqueueNotification(Order* order) {
        pendingNotifications.push(order);
    }
    void processAsync() {
        // Run in a background thread
        while(isProcessing) {
            if(!pendingNotifications.empty()) {
                Order* order = pendingNotifications.front();
                pendingNotifications.pop();
                if(!sendNotification(order)) {
                    // Retry on failure
                    enqueueNotification(order);
                }
            }
        }
    }
};
